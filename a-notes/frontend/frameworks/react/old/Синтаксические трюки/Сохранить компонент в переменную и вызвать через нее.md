```tsx
import { BookCardDefault } from './BookCardDefault';
import { BookCardCompact } from './BookCardCompact';
import { BookCardDetailed } from './BookCardDetailed';
import { BookCardProps } from './types';


export default function BookCard({ book, variant = 'default' }: BookCardProps) {
  const components = {
    default: BookCardDefault,
    compact: BookCardCompact,
    detailed: BookCardDetailed
  };

  const Component = components[variant];  // <-- Сохраняем компонент в переменную
  return <Component book={book} />;  // <-- Вызываем компонент через нее
}
```

Здесь показано как обойтись без if'ов в ситуации, когда надо вызвать один из компонентов в зависимости от условия.

Поскольку jsx преобразуется в React.createElement, то конструкция `return <Component book={book} />` превратится в такое:

```tsx
return React.createElement(BookCardCompact, { book: book });
```
